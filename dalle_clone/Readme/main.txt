sec1 - setup react client
~~~~~~~~~~~~~~~~~~~~~~~~~
- npm create vite@latest ./
- npm install file-saver
- tailwindcss:
    npm install -D tailwindcss postcss autoprefixer
    npx tailwindcss init -p
    update tailwind.config.js file
    update index.css
- npm run dev
